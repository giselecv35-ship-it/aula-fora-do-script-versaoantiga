      // Função Typewriter FIXED FOR MOBILE (suporta quebra de linha)
      function typeWriter(element, text, speed = 50) {
        let i = 0;
        element.innerHTML = '';
        function type() {
          if (i < text.length) {
            element.innerHTML += text.charAt(i);
            i++;
            setTimeout(type, speed);
          }
        }
        type();
      }

      // Intersection Observer para animações de fade e digitação
      const observer = new IntersectionObserver((entries) => {
        entries.forEach(entry => {
          if (entry.isIntersecting) {
            entry.target.classList.add('visible');
            
            // Verifica se o elemento visível contém o typewriter
            const typingEl = entry.target.querySelector('[data-typewriter-text]');
            if (typingEl && !typingEl.dataset.started) {
                typingEl.dataset.started = 'true';
                typeWriter(typingEl, typingEl.dataset.typewriterText);
            }
          }
        });
      }, { threshold: 0.1 });

      document.querySelectorAll('.fade-in-up').forEach(el => observer.observe(el));
      
      const typingTrigger = document.getElementById('typing-trigger');
      if (typingTrigger) observer.observe(typingTrigger);

      // Imagens do Carrossel
      const images = [
          'https://i.postimg.cc/LsVDGFcV/1.jpg',

          'https://i.postimg.cc/k5ysLP0H/5.jpg',
          'https://i.postimg.cc/vmvhCwJt/6.jpg'
      ];

      // Lightbox Logic
      let currentIdx = 0;
      const lightbox = document.getElementById('lightbox');
      const lightboxImg = document.getElementById('lightbox-img');

      function openLightbox(index) {
          currentIdx = index;
          lightboxImg.src = images[currentIdx];
          lightbox.classList.remove('hidden');
          setTimeout(() => lightbox.classList.add('active'), 10);
          document.body.style.overflow = 'hidden';
      }

      function closeLightbox() {
          lightbox.classList.remove('active');
          setTimeout(() => {
            lightbox.classList.add('hidden');
            document.body.style.overflow = '';
          }, 300);
      }

      function nextLightbox() {
          currentIdx = (currentIdx + 1) % images.length;
          updateLightboxImage();
      }

      function prevLightbox() {
          currentIdx = (currentIdx - 1 + images.length) % images.length;
          updateLightboxImage();
      }

      function updateLightboxImage() {
          lightboxImg.style.opacity = 0;
          lightboxImg.style.transform = 'scale(0.95)';
          setTimeout(() => {
              lightboxImg.src = images[currentIdx];
              lightboxImg.style.opacity = 1;
              lightboxImg.style.transform = 'scale(1)';
          }, 150);
      }

      // Keyboard listeners
      document.addEventListener('keydown', (e) => {
          if (!lightbox.classList.contains('hidden')) {
              if (e.key === 'Escape') closeLightbox();
              if (e.key === 'ArrowRight') nextLightbox();
              if (e.key === 'ArrowLeft') prevLightbox();
          }
      });

      // Carousel Scroll Logic
      const carousel = document.getElementById('image-carousel');
      const dots = document.querySelectorAll('.indicator-dot');
      const nextBtn = document.getElementById('nextBtn');
      const prevBtn = document.getElementById('prevBtn');

      function updateDots(index) {
          dots.forEach((dot, i) => {
              if (i === index) {
                  dot.classList.add('active-dot');
                  dot.classList.remove('bg-text-muted');
              } else {
                  dot.classList.remove('active-dot');
                  dot.classList.add('bg-text-muted');
              }
          });
      }

      function scrollCarousel(direction) {
          const itemWidth = carousel.querySelector('.carousel-item').offsetWidth + 24; 
          carousel.scrollBy({ left: direction * itemWidth, behavior: 'smooth' });
      }

      nextBtn.addEventListener('click', () => scrollCarousel(1));
      prevBtn.addEventListener('click', () => scrollCarousel(-1));

      carousel.addEventListener('scroll', () => {
          const itemWidth = carousel.querySelector('.carousel-item').offsetWidth + 24;
          const index = Math.round(carousel.scrollLeft / itemWidth);
          if(index >= 0 && index < dots.length) updateDots(index);
      });

      dots.forEach((dot, i) => {
          dot.addEventListener('click', () => {
              const itemWidth = carousel.querySelector('.carousel-item').offsetWidth + 24;
              carousel.scrollTo({ left: i * itemWidth, behavior: 'smooth' });
          });
      });
